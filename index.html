// 🌟 韋杰本本專屬後台大腦 (支援紀錄修改、舊照片保留) var SHEET_URL = 'https://docs.google.com/spreadsheets/d/1Ld9KAxbg4WEwylbDqz6ZRxt2YazfKNQR-jIlQWiiVJA/edit?usp=drive_link'; var DRIVE_FOLDER_ID = '1ugpwqi2cgEpu0ODfewS79rRHfZzMsIsS'; function doPost(e) { try { var msg = JSON.parse(e.postData.contents); if (msg.source === 'webapp') { if (msg.action === 'submit') return ContentService.createTextOutput(processForm(msg.data)).setMimeType(ContentService.MimeType.TEXT); if (msg.action === 'deleteRow') return ContentService.createTextOutput(deleteByRowWeb(msg.data.row)).setMimeType(ContentService.MimeType.TEXT); if (msg.action === 'getHistory') return ContentService.createTextOutput(JSON.stringify(getCloudHistory())).setMimeType(ContentService.MimeType.JSON); if (msg.action === 'getItems') return ContentService.createTextOutput(JSON.stringify(getAutocompleteItems())).setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput("OK").setMimeType(ContentService.MimeType.TEXT); } catch (err) { return ContentService.createTextOutput("Error: " + err.toString()).setMimeType(ContentService.MimeType.TEXT); } } function processForm(data) { var sheet = SpreadsheetApp.openByUrl(SHEET_URL).getActiveSheet(); var amountData = parseAmountData(data.amount); if (!amountData) return "❌ 錯誤：找不到金額。"; var fileUrl = ""; // 🌟 如果是編輯模式且沒有上傳新照片，自動抓取原本試算表上的照片連結 var oldFileUrl = ""; if (data.editRow && parseInt(data.editRow) > 1) { oldFileUrl = sheet.getRange(parseInt(data.editRow), 7).getValue(); } if (data.photoBase64) { var blob = Utilities.newBlob(Utilities.base64Decode(data.photoBase64.split(',')[1]), "image/jpeg", "收據_" + Utilities.formatDate(new Date(), "GMT+8", "yyyyMMdd_HHmmss") + ".jpg"); fileUrl = DriveApp.getFolderById(DRIVE_FOLDER_ID).createFile(blob).getUrl(); } else if (oldFileUrl && oldFileUrl.indexOf("http") !== -1) { fileUrl = oldFileUrl; // 沿用舊照片 } var recordTimeStr = data.time ? data.time.replace('T', ' ') : Utilities.formatDate(new Date(), "GMT+8", "yyyy-MM-dd HH:mm"); var recordType = data.type, recordItem = data.item, recordAmount = amountData.twdAmount; var payment = (data.payment && data.payment.trim() !== "") ? data.payment.trim() : "現金"; var baseNote = data.note || ""; var sheetPayer = (data.payer && data.payer.trim() !== "") ? data.payer.trim() : "Luke"; var returnMsg = ""; var sheetRowData = [ "'" + recordTimeStr, recordType, "'" + recordItem, recordAmount, payment, "", fileUrl, sheetPayer ]; if (data.type === "墊付") { var splits = data.splits || []; if (splits.length > 0) { var totalSplitTwdAmt = 0; var totalSplitOriginAmt = 0; var splitDetails = []; var isSelfPayer = (sheetPayer === "自己" || sheetPayer === "Luke" || sheetPayer === "杰" || sheetPayer.toUpperCase() === "LUKE"); for (var j = 0; j < splits.length; j++) { var pName = splits[j].name.trim(); var pAmount = splits[j].amount; if (isSelfPayer && (pName.toUpperCase() === "LUKE" || pName === "自己" || pName === "杰")) continue; var cellText = ""; var twdAmt = 0; if (amountData.currency !== "TWD") { twdAmt = Math.round(pAmount * amountData.rate); cellText = pName + "(" + pAmount + amountData.currency + "/$" + twdAmt + ")"; totalSplitOriginAmt += pAmount; } else { twdAmt = pAmount; cellText = pName + "($" + twdAmt + ")"; } sheetRowData.push(cellText); splitDetails.push(cellText); totalSplitTwdAmt += twdAmt; } var myShareTwd = recordAmount - totalSplitTwdAmt; var myShareOrigin = amountData.originAmount - totalSplitOriginAmt; var splitSummaryText = ""; var targetListText = splitDetails.length > 0 ? "代墊給：" + splitDetails.join(", ") : ""; if (!isSelfPayer) { sheetRowData[1] = "純記帳"; sheetRowData[3] = 0; splitSummaryText = ""; returnMsg = "⚖️ 幫朋友記帳完成！\n總額：" + (amountData.currency !== "TWD" ? amountData.originAmount + " " + amountData.currency + " (折合台幣 $" + recordAmount + ")" : "$" + recordAmount) + (targetListText ? "\n" + targetListText : ""); } else { if (amountData.currency !== "TWD") { splitSummaryText = "自己負擔外幣: " + myShareOrigin + " " + amountData.currency + " (折合台幣 $" + myShareTwd + ")"; } else { splitSummaryText = "自己負擔：$" + myShareTwd; } returnMsg = "⚖️ 墊付計算完成！\n總額：" + (amountData.currency !== "TWD" ? amountData.originAmount + " " + amountData.currency + " (折合台幣 $" + recordAmount + ")" : "$" + recordAmount) + (targetListText ? "\n" + targetListText : "") + "\n" + splitSummaryText; } var finalNote = splitSummaryText; if (amountData.note) finalNote = finalNote ? amountData.note + " | " + finalNote : amountData.note; if (baseNote) finalNote = finalNote ? finalNote + " | " + baseNote : baseNote; sheetRowData[5] = "'" + finalNote; } } else { var finalNote = amountData.note || ""; if (baseNote) finalNote = finalNote ? finalNote + " | " + baseNote : baseNote; sheetRowData[5] = "'" + finalNote; returnMsg = (amountData.note ? "💱 匯率換算成功！\n" : "") + "✅ 紀錄成功：$" + recordAmount; } // 🌟 核心：判定是修改還是新增 if (data.editRow && parseInt(data.editRow) > 1) { var targetRow = parseInt(data.editRow); var maxCol = Math.max(8, sheet.getLastColumn()); sheet.getRange(targetRow, 1, 1, maxCol).clearContent(); // 先把整排淨空，避免舊的代墊人資料殘留 sheet.getRange(targetRow, 1, 1, sheetRowData.length).setValues([sheetRowData]); // 讓修改提示更精簡 returnMsg = "✏️ 紀錄已成功更新並取代舊資料！\n" + returnMsg.replace(/✅ 紀錄成功：/g, '').replace(/⚖️ 墊付計算完成！\n/g, '').replace(/⚖️ 幫朋友記帳完成！\n/g, '').trim(); } else { sheet.appendRow(sheetRowData); } if (data.photoBase64 && !oldFileUrl) returnMsg += "\n📸 已附新照片"; else if (oldFileUrl) returnMsg += "\n🖼️ (保留原照片)"; return returnMsg; } function getAutocompleteItems() { var sheet = SpreadsheetApp.openByUrl(SHEET_URL).getActiveSheet(); var lastRow = sheet.getLastRow(); var lastCol = Math.max(8, sheet.getLastColumn()); if (lastRow <= 1) return { items: [], payers: [], payments: [] }; var data = sheet.getRange(Math.max(2, lastRow - 299), 1, Math.min(lastRow - 1, 300), lastCol).getValues(); var uniqueItems = [], uniquePayers = [], uniquePayments = []; var seenI = {}, seenP = {}, seenPay = {}; for (var i = data.length - 1; i >= 0; i--) { var itm = data[i][2] ? data[i][2].toString().trim() : ""; var pay = data[i][4] ? data[i][4].toString().trim() : ""; if (itm && itm.indexOf("純記帳") === -1 && !seenI[itm]) { seenI[itm] = true; uniqueItems.push(itm); } if (pay && pay !== "現金" && !seenPay[pay]) { seenPay[pay] = true; uniquePayments.push(pay); } var mainPayer = data[i][7] ? data[i][7].toString().trim() : ""; if (mainPayer && mainPayer !== "自己" && mainPayer !== "Luke" && mainPayer !== "杰" && !seenP[mainPayer]) { seenP[mainPayer] = true; uniquePayers.push(mainPayer); } for (var colIdx = 8; colIdx < lastCol; colIdx++) { var cellData = data[i][colIdx] ? data[i][colIdx].toString().trim() : ""; if (cellData) { var match = cellData.match(/^(.*?)\(/); var pyr = match ? match[1].trim() : cellData; if (pyr && pyr !== "自己" && pyr !== "Luke" && pyr !== "杰" && !seenP[pyr]) { seenP[pyr] = true; uniquePayers.push(pyr); } } } } return { items: uniqueItems, payers: uniquePayers, payments: uniquePayments }; } // 🌟 核心：將整組 Raw Data 傳給前端當成還原資料的素材庫 function getCloudHistory() { var sheet = SpreadsheetApp.openByUrl(SHEET_URL).getActiveSheet(); var lastRow = sheet.getLastRow(), startRow = Math.max(2, lastRow - 19); var lastCol = Math.max(8, sheet.getLastColumn()); if (lastRow <= 1) return []; var data = sheet.getRange(startRow, 1, (lastRow - startRow + 1), lastCol).getValues(); var result = []; for (var i = 0; i < data.length; i++) { var row = data[i]; if (!row[0]) continue; result.push({ rowNum: startRow + i, rowData: row, // 傳送整排完整原始資料給前端解析 timeStr: Utilities.formatDate(new Date(row[0]), "GMT+8", "MM/dd HH:mm"), text: row[1] + "：" + row[2] + " $" + row[3] }); } return result.reverse(); } function deleteByRowWeb(rowNum) { var sheet = SpreadsheetApp.openByUrl(SHEET_URL).getActiveSheet(); if (rowNum > sheet.getLastRow() || rowNum < 2) return "⚠️ 資料已不存在。"; var item = sheet.getRange(rowNum, 3).getValue(), photoUrl = sheet.getRange(rowNum, 7).getValue(); if (photoUrl && photoUrl.indexOf("drive") !== -1) { try { DriveApp.getFileById(photoUrl.match(/[-\w]{25,}/)[0]).setTrashed(true); } catch(e) {} } sheet.deleteRow(rowNum); return "🗑️ 刪除成功：" + item; } function parseAmountData(rawStr) { if (!rawStr) return null; var str = rawStr.toString().toUpperCase().replace(/\s/g, ''); var currencyMatch = str.match(/[A-Z]+/); var currency = currencyMatch ? currencyMatch[0] : "TWD"; var amount = parseFloat(str.replace(/[A-Z]/g, '')); if (isNaN(amount)) return null; if (currency === "TWD") return { twdAmount: amount, currency: "TWD", rate: 1, originAmount: amount, note: "" }; try { var rate = JSON.parse(UrlFetchApp.fetch("https://open.er-api.com/v6/latest/" + currency).getContentText()).rates["TWD"]; return { twdAmount: Math.round(amount * rate), currency: currency, rate: rate, originAmount: amount, note: "外幣: " + amount + " " + currency + " (匯率" + rate.toFixed(4) + ")" }; } catch (e) { return { twdAmount: amount, currency: currency, rate: 1, originAmount: amount, note: "(換算失敗)" }; } }
